//
// DOCUMENTO ÚNICO: chat/sala_principal
// - Qualquer um autenticado (incluindo anônimo) pode ler e escrever
// - Valida tamanho máximo do array (50 msgs) e campos obrigatórios

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ── Regras existentes do sistema ──────────────────────────────────────
    // (mantenha aqui o resto das suas regras)

    // ── Chat: documento único ─────────────────────────────────────────────
    match /chat/sala_principal {

      // Qualquer usuário autenticado (incluindo anônimo via signInAnonymously) pode ler
      allow read: if request.auth != null;

      // Escrita: autenticado + validação de estrutura
      allow write: if request.auth != null
        && request.resource.data.keys().hasOnly(['msgs'])
        && request.resource.data.msgs is list
        && request.resource.data.msgs.size() <= 50;
    }

  }
}
